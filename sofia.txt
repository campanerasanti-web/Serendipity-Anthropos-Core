import { useEffect, useRef, useState } from "react";
import { Html5Qrcode } from "html5-qrcode";
import { Button } from "@/components/ui/button";
import { Camera, X, CheckCircle2, Zap } from "lucide-react";
import { useTranslation } from "react-i18next";
import { toast } from "sonner"; // Usamos sonner para un feedback m√°s elegante

interface QrScannerProps {
  onScan: (result: string) => void;
}

export default function QrScanner({ onScan }: QrScannerProps) {
  const { t, i18n } = useTranslation();
  const [scanning, setScanning] = useState(false);
  const [lastScan, setLastScan] = useState<string | null>(null);
  const scannerRef = useRef<Html5Qrcode | null>(null);
  const containerId = "qr-reader";

  const startScanning = async () => {
    setScanning(true);
    setLastScan(null);
    
    try {
      const scanner = new Html5Qrcode(containerId);
      scannerRef.current = scanner;
      
      // Configuraci√≥n optimizada para la planta (Vietnam Industrial Setting)
      const config = {
        fps: 15, // Mayor FPS para capturas r√°pidas
        qrbox: { width: 280, height: 280 },
        aspectRatio: 1.0
      };

      await scanner.start(
        { facingMode: "environment" },
        config,
        (decodedText) => {
          handleSuccess(decodedText);
        },
        () => {} // Errores silenciosos para no interrumpir el flujo
      );
    } catch (err) {
      console.error("QR scanner error:", err);
      toast.error(t("errors.camera_access"));
      setScanning(false);
    }
  };

  const handleSuccess = (decodedText: string) => {
    setLastScan(decodedText);
    onScan(decodedText);
    
    // Feedback Humano y Profesional
    const successMsg = i18n.language === 'vi' 
      ? "L√†m t·ªët l·∫Øm! T√†i li·ªáu ƒë√£ ƒë∆∞·ª£c ghi l·∫°i." // "¬°Buen trabajo! Material registrado."
      : "¬°Maestr√≠a lograda! Material registrado correctamente.";
    
    toast.success(successMsg, {
      icon: <CheckCircle2 className="h-5 w-5 text-green-500" />,
      duration: 3000,
    });

    stopScanning();
  };

  const stopScanning = async () => {
    try {
      if (scannerRef.current?.isScanning) {
        await scannerRef.current.stop();
      }
      scannerRef.current?.clear();
    } catch (e) {
      console.warn("Clean up error", e);
    }
    scannerRef.current = null;
    setScanning(false);
  };

  useEffect(() => {
    return () => { stopScanning(); };
  }, []);

  return (
    <div className="space-y-4">
      {!scanning ? (
        <div className="relative group">
          <div className="absolute -inset-0.5 bg-gradient-to-r from-primary to-accent rounded-lg blur opacity-30 group-hover:opacity-100 transition duration-1000"></div>
          <Button 
            onClick={startScanning} 
            className="relative w-full h-16 text-lg font-display bg-background border-2 border-primary/20 hover:border-primary transition-all"
          >
            <Camera className="h-6 w-6 mr-3 text-primary" />
            {t("measurement.scanQr")}
          </Button>
          <p className="text-[10px] text-center mt-2 text-muted-foreground uppercase tracking-tighter italic">
            {t("measurement.operator_instruction")}
          </p>
        </div>
      ) : (
        <div className="space-y-4 animate-in fade-in zoom-in duration-300">
          <div className="relative">
            <div id={containerId} className="rounded-2xl overflow-hidden border-4 border-primary/30 shadow-2xl" />
            <div className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 w-48 h-48 border-2 border-primary/50 rounded-lg pointer-events-none animate-pulse">
               <Zap className="absolute -top-6 -left-6 text-primary h-6 w-6" />
            </div>
          </div>
          
          <Button variant="ghost" size="sm" onClick={stopScanning} className="w-full text-destructive hover:bg-destructive/10">
            <X className="h-4 w-4 mr-2" />
            {t("measurement.stopScan")}
          </Button>
        </div>
      )}
    </div>
  );
}

const RESERVE_GOAL_VND = 1000000000; 

export const SurvivalThermometer = () => {
  const { t } = useTranslation();
  const [data, setData] = useState({
    totalInvoiced: 0,
    monthlyCosts: 0,
    reserveFund: 0,
  });

  useEffect(() => {
    const fetchFinancialStats = async () => {
      const now = new Date();
      // 1. Obtener Costos Fijos Reales del mes
      const { data: costs } = await supabase
        .from("fixed_costs")
        .select("*")
        .eq("month", now.getMonth() + 1)
        .eq("year", now.getFullYear())
        .maybeSingle();

      // 2. Obtener Facturaci√≥n Total del mes
      const { data: inv } = await supabase.from("invoices").select("total_amount");
      const totalInvoiced = inv?.reduce((acc, curr) => acc + curr.total_amount, 0) || 0;

      const payroll = costs?.payroll || 0;
      const rent = costs?.rent || 0;
      const evn = costs?.evn || 0;
      const other = costs?.other_costs || 0;

      setData({
        totalInvoiced,
        monthlyCosts: payroll + rent + evn + other,
        reserveFund: totalInvoiced > (payroll + rent + evn + other) 
                     ? (totalInvoiced - (payroll + rent + evn + other)) * 0.10 // 10% al fondo
                     : 0
      });
    };

    fetchFinancialStats();
  }, []);

  const breakEven = data.monthlyCosts || 20000; // Default a 20k si no hay datos
  const progress = Math.min((data.totalInvoiced / breakEven) * 100, 100);
  const status = progress < 100 ? "yellow" : "green";

  return (
    <Card className={`border-l-4 ${status === 'green' ? 'border-l-green-500' : 'border-l-yellow-500'} bg-card/50 backdrop-blur`}>
      <CardContent className="pt-6">
        <div className="flex justify-between items-end mb-4">
          <div>
            <p className="text-sm font-medium text-muted-foreground uppercase tracking-wider">
              {t("status.survival_mode")}
            </p>
            <h2 className="text-3xl font-bold font-display">
              ${data.totalInvoiced.toLocaleString()} 
              <span className="text-lg text-muted-foreground font-normal"> / ${breakEven.toLocaleString()}</span>
            </h2>
          </div>
          <div className="text-right">
             {progress >= 100 ? (
               <div className="flex items-center text-green-500 font-bold">
                 <Shield className="mr-2 h-5 w-5" /> {t("status.peace")}
               </div>
             ) : (
               <div className="flex items-center text-yellow-500 font-bold animate-pulse">
                 <AlertCircle className="mr-2 h-5 w-5" /> {t("status.working")}
               </div>
             )}
          </div>
        </div>

        <Progress value={progress} className="h-3 mb-2" />
        
        <div className="grid grid-cols-2 gap-4 mt-6">
          <div className="bg-primary/5 p-3 rounded-lg border border-primary/10">
            <p className="text-xs text-muted-foreground flex items-center">
              <TrendingUp className="h-3 w-3 mr-1" /> {t("financials.net_flow")}
            </p>
            <p className="text-lg font-bold">${(data.totalInvoiced - data.monthlyCosts).toLocaleString()}</p>
          </div>
          <div className="bg-green-500/5 p-3 rounded-lg border border-green-500/10">
            <p className="text-xs text-muted-foreground flex items-center">
              <Heart className="h-3 w-3 mr-1" /> {t("financials.peace_fund")}
            </p>
            <p className="text-lg font-bold text-green-600">${data.reserveFund.toLocaleString()}</p>
          </div>
        </div>
        
        <p className="mt-4 text-xs italic text-muted-foreground text-center">
          "Administrando con sabidur√≠a lo que el Padre provee hoy."
        </p>
      </CardContent>
    </Card>
  );
};
